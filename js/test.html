input program

{
	# Dyck's language
	@string str "(()(()))".
	S => '(' S ')' S | null.
}
{
	# create graph represented by relation e
	e(v1 v2).  # with nodes: v1 -> v2,
	e(v2 v3).  # v2 -> v3
	e(v3 v1).  # v3 -> v1

	# transitive closure
	tc(?x ?y) :- e(?x ?y).
	tc(?x ?y) :- tc(?x ?z), tc(?z ?y).
}
{
	~S(?x ?x) :- S(?x ?x).
	~e(?x ?y) :- e(?x ?y).
}
